    <script>
        let uploadedFile = null;
        let lastClassificationResult = null;

        // Initialize
        document.addEventListener('DOMContentLoaded', function() {
            loadStats();
            
            const uploadZone = document.getElementById('uploadZone');
            const fileInput = document.getElementById('fileInput');

            uploadZone.addEventListener('click', () => fileInput.click());

            fileInput.addEventListener('change', (e) => {
                if (e.target.files.length > 0) {
                    handleFile(e.target.files[0]);
                }
            });

            uploadZone.addEventListener('dragover', (e) => {
                e.preventDefault();
                uploadZone.classList.add('dragover');
            });

            uploadZone.addEventListener('dragleave', () => {
                uploadZone.classList.remove('dragover');
            });

            uploadZone.addEventListener('drop', (e) => {
                e.preventDefault();
                uploadZone.classList.remove('dragover');
                if (e.dataTransfer.files.length > 0) {
                    handleFile(e.dataTransfer.files[0]);
                }
            });
        });

        function handleFile(file) {
            if (!file.type.startsWith('image/')) {
                showToast('Please upload an image file', 'error');
                return;
            }

            uploadedFile = file;
            
            const reader = new FileReader();
            reader.onload = (e) => {
                document.getElementById('previewImage').src = e.target.result;
                document.getElementById('previewContainer').style.display = 'block';
                document.getElementById('resultPlaceholder').style.display = 'block';
                document.getElementById('resultContent').style.display = 'none';
            };
            reader.readAsDataURL(file);

            showToast('Image loaded successfully!', 'success');
        }

        async function classifyImage() {
            if (!uploadedFile) {
                showToast('Please upload an image first', 'error');
                return;
            }

            document.getElementById('loadingOverlay').style.display = 'flex';
            document.getElementById('classifyBtn').disabled = true;

            const formData = new FormData();
            formData.append('file', uploadedFile);

            try {
                const response = await fetch('/api/classify', {
                    method: 'POST',
                    body: formData
                });

                const result = await response.json();

                if (result.success) {
                    lastClassificationResult = result;
                    displayResult(result);
                    loadStats();
                    showToast('Classification completed!', 'success');
                } else {
                    showToast(`Error: ${result.error}`, 'error');
                }
            } catch (error) {
                showToast(`Error: ${error.message}`, 'error');
            } finally {
                document.getElementById('loadingOverlay').style.display = 'none';
                document.getElementById('classifyBtn').disabled = false;
            }
        }

        function displayResult(result) {
            const isBio = result.classification === 'biodegradable';
            const confidence = (result.confidence * 100).toFixed(1);

            let html = `
                <div class="result-container ${isBio ? 'result-biodegradable' : 'result-non-biodegradable'}">
                    <div class="result-header">
                        <i class="fas fa-${isBio ? 'leaf' : 'times-circle'}"></i>
                        ${result.classification.toUpperCase()}
                    </div>

                    <div class="confidence-bar">
                        <div class="confidence-fill" style="width: ${confidence}%">
                            ${confidence}% Confidence
                        </div>
                    </div>

                    <div class="details-grid">
                        <div class="detail-item">
                            <div class="detail-label"><i class="fas fa-box"></i> Material Type</div>
                            <div class="detail-value">${result.material_type || 'Unknown'}</div>
                        </div>
                        <div class="detail-item">
                            <div class="detail-label"><i class="fas fa-robot"></i> Method</div>
                            <div class="detail-value">${result.method || 'AI'}</div>
                        </div>
                        <div class="detail-item">
                            <div class="detail-label"><i class="fas fa-trash-alt"></i> Bin Color</div>
                            <div class="detail-value">${result.recommendations?.bin_color || 'N/A'}</div>
                        </div>
                        <div class="detail-item">
                            <div class="detail-label"><i class="fas fa-clock"></i> Decomposition</div>
                            <div class="detail-value">${result.recommendations?.decomposition_time || 'N/A'}</div>
                        </div>
                    </div>

                    ${result.recommendations ? `
                        <div class="recommendations">
                            <h4><i class="fas fa-lightbulb"></i> Disposal Recommendations</h4>
                            <ul>
                                ${result.recommendations.tips.map(tip => `
                                    <li><i class="fas fa-check-circle" style="color: #28a745;"></i> ${tip}</li>
                                `).join('')}
                            </ul>
                        </div>
                    ` : ''}

                    ${result.top_predictions ? `
                        <div class="predictions-list">
                            <h5 style="color: #333; margin-top: 20px;">Top Predictions:</h5>
                            ${result.top_predictions.slice(0, 5).map((pred, idx) => `
                                <div class="prediction-item">
                                    <span>${idx + 1}. ${pred.label}</span>
                                    <span style="font-weight: 700;">${(pred.confidence * 100).toFixed(1)}%</span>
                                </div>
                            `).join('')}
                        </div>
                    ` : ''}

                    <div style="text-align: center; margin-top: 20px;">
                        <button class="btn btn-primary" onclick="exportSinglePDF()">
                            <i class="fas fa-file-pdf"></i> Export This Report as PDF
                        </button>
                    </div>
                </div>
            `;

            document.getElementById('resultPlaceholder').style.display = 'none';
            document.getElementById('resultContent').innerHTML = html;
            document.getElementById('resultContent').style.display = 'block';
        }

        async function loadStats() {
            try {
                const response = await fetch('/api/stats');
                const data = await response.json();
                
                if (data.success) {
                    document.getElementById('totalClassifications').textContent = data.stats.total_classifications;
                    document.getElementById('biodegradableCount').textContent = data.stats.biodegradable;
                    document.getElementById('nonBiodegradableCount').textContent = data.stats.non_biodegradable;
                    document.getElementById('modelStatus').textContent = data.stats.system_status === 'deep_learning' ? 'Active' : 'Backup';
                }
            } catch (error) {
                console.error('Error loading stats:', error);
            }
        }

        async function clearHistory() {
            if (confirm('Are you sure you want to clear all classification history?')) {
                try {
                    const response = await fetch('/api/clear_history', { method: 'POST' });
                    const data = await response.json();
                    
                    if (data.success) {
                        loadStats();
                        showToast('History cleared successfully', 'success');
                    }
                } catch (error) {
                    showToast(`Error: ${error.message}`, 'error');
                }
            }
        }

        function resetInterface() {
            document.getElementById('previewContainer').style.display = 'none';
            document.getElementById('resultPlaceholder').style.display = 'block';
            document.getElementById('resultContent').style.display = 'none';
            document.getElementById('fileInput').value = '';
            uploadedFile = null;
            lastClassificationResult = null;
            showToast('Interface reset', 'info');
        }

        // Export Functions
        async function exportPDF() {
            try {
                showToast('Generating PDF report...', 'info');
                
                const response = await fetch('/api/export/pdf', {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({type: 'history'})
                });

                if (!response.ok) throw new Error('PDF generation failed');

                const blob = await response.blob();
                const url = window.URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = `waste_classification_report_${Date.now()}.pdf`;
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                window.URL.revokeObjectURL(url);

                showToast('PDF report downloaded!', 'success');
            } catch (error) {
                showToast(`Error: ${error.message}`, 'error');
            }
        }

        async function exportSinglePDF() {
            if (!lastClassificationResult) {
                showToast('No classification result to export', 'error');
                return;
            }

            try {
                showToast('Generating PDF report...', 'info');
                
                const response = await fetch('/api/export/pdf', {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({
                        type: 'single',
                        result: lastClassificationResult
                    })
                });

                if (!response.ok) throw new Error('PDF generation failed');

                const blob = await response.blob();
                const url = window.URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = `classification_report_${Date.now()}.pdf`;
                document.body.appendChild(a);
                a.click();
                document.body.removeChild(a);
                window.URL.revokeObjectURL(url);

                showToast('PDF report downloaded!', 'success');
            } catch (error) {
                showToast(`Error: ${error.message}`, 'error');
            }
        }

        async function exportJSON() {
            try {
                showToast('Exporting JSON data...', 'info');
                
                const response = await fetch('/api/export/json');
                if (!response.ok) throw new Error('JSON export failed');

                const blob = await response.blob();
                const url = window.URL.createObjectURL(blob);
                const a = document.createElement('a');
                a.href = url;
                a.download = `waste_classification_data_${Date.now()}.json`;

        function showToast(message, type = 'info') {
            const colors = {
                success: '#28a745',
                error: '#dc3545',
                info: '#17a2b8',
                warning: '#ffc107'
            };

            const icons = {
                success: 'check-circle',
                error: 'times-circle',
                info: 'info-circle',
                warning: 'exclamation-triangle'
            };

            const toast = document.createElement('div');
            toast.className = 'toast';
            toast.style.borderLeft = `5px solid ${colors[type]}`;
            toast.innerHTML = `
                <div style="display: flex; align-items: center; gap: 10px;">
                    <i class="fas fa-${icons[type]}" style="color: ${colors[type]}; font-size: 1.5rem;"></i>
                    <span style="font-weight: 600;">${message}</span>
                </div>
            `;

            document.getElementById('toastContainer').appendChild(toast);

            setTimeout(() => {
                toast.style.animation = 'slideInRight 0.3s ease reverse';
                setTimeout(() => toast.remove(), 300);
            }, 3000);
        }
