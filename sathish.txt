# ADVANCED IMAGE PREPROCESSING
# ============================================================================
def preprocess_image_for_model(image_path):
    try:
        img = keras_image.load_img(image_path, target_size=(224, 224))
        img_array = keras_image.img_to_array(img)
        img_array = np.expand_dims(img_array, axis=0)
        img_array = preprocess_input(img_array)
        return img_array
    except Exception as e:
        logger.error(f"ERROR: Image preprocessing failed: {e}")
        return None

# ============================================================================
# DEEP LEARNING CLASSIFICATION
# ============================================================================
def deep_learning_classify(image_path):
    logger.info("DEEP LEARNING: Analyzing image with MobileNetV2...")
    
    try:
        processed_image = preprocess_image_for_model(image_path)
        if processed_image is None:
            raise ValueError("Image preprocessing failed")
        
        predictions = base_model.predict(processed_image)
        decoded_predictions = decode_predictions(predictions, top=10)[0]
        
        logger.info("TOP 10 PREDICTIONS:")
        for i, (imagenet_id, label, score) in enumerate(decoded_predictions, 1):
            logger.info(f"  {i}. {label}: {score*100:.2f}%")
        
        is_biodegradable = False
        confidence = 0.0
        detected_objects = []
        reasoning = []
        
        for imagenet_id, label, score in decoded_predictions:
            label_lower = label.lower().replace('_', ' ')
            detected_objects.append({'label': label, 'confidence': float(score)})
            
            for keyword in BIODEGRADABLE_KEYWORDS:
                if keyword in label_lower:
                    is_biodegradable = True
                    confidence = max(confidence, float(score) * 0.95)
                    reasoning.append(f"Detected: {label} (biodegradable item)")
                    break
            
            for keyword in NON_BIODEGRADABLE_KEYWORDS:
                if keyword in label_lower:
                    is_biodegradable = False
                    confidence = max(confidence, float(score) * 0.95)
                    reasoning.append(f"Detected: {label} (non-biodegradable item)")
                    break
        
        if confidence < 0.5:
            logger.info("LOW CONFIDENCE: Using visual analysis backup...")
            visual_result = advanced_visual_analysis(image_path)
            is_biodegradable = visual_result['is_biodegradable']
            confidence = max(confidence, visual_result['confidence'])
            reasoning.append(f"Visual analysis: {visual_result['material_type']}")
        
        top_prediction = decoded_predictions[0]
        result = {
            'success': True,
            'classification': 'biodegradable' if is_biodegradable else 'non-biodegradable',
            'confidence': confidence,
            'detected_object': top_prediction[1],
            'material_type': top_prediction[1],
            'method': 'deep_learning_mobilenetv2',
            'top_predictions': detected_objects[:5],
            'reasoning': reasoning,
            'ai_analysis': {
                'model': 'MobileNetV2',
                'framework': 'TensorFlow/Keras',
                'weights': 'ImageNet',
                'total_predictions': len(decoded_predictions)
            },
            'recommendations': get_recommendations(is_biodegradable)
        }
        
        logger.info(f"SUCCESS: Classification: {result['classification']} ({confidence*100:.1f}%)")
        return result
        
    except Exception as e:
        logger.error(f"ERROR: Deep learning classification failed: {e}")
        traceback.print_exc()
        return fallback_visual_classify(image_path)

# ============================================================================
# ADVANCED VISUAL ANALYSIS
# ============================================================================
def advanced_visual_analysis(image_path):
    try:
        image = cv2.imread(image_path)
        hsv = cv2.cvtColor(image, cv2.COLOR_BGR2HSV)
        avg_hue = np.mean(hsv[:,:,0])
        avg_sat = np.mean(hsv[:,:,1])
        avg_val = np.mean(hsv[:,:,2])
        
        gray = cv2.cvtColor(image, cv2.COLOR_BGR2GRAY)
        variance = np.var(gray)
        edges = cv2.Canny(gray, 50, 150)
        edge_density = np.sum(edges > 0) / edges.size
        contours, _ = cv2.findContours(edges, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
        num_contours = len(contours)
        brightness = np.mean(gray)
        
        is_biodegradable = False
        confidence = 0.0
        material_type = "Unknown"
        
        if (20 <= avg_hue <= 90 and avg_sat > 40) or (10 <= avg_hue <= 30 and avg_val < 180):
            if variance > 800 and edge_density > 0.05:
                is_biodegradable = True
                confidence = 0.78
                material_type = "Organic/Natural Material"
        elif avg_sat < 50 and variance < 500 and edge_density < 0.08:
            is_biodegradable = False
            confidence = 0.82
            material_type = "Synthetic/Plastic Material"
        elif brightness > 150 and avg_sat < 40 and edge_density > 0.15:
            is_biodegradable = False
            confidence = 0.85
            material_type = "Metal/Glass"
        elif 10 <= avg_hue <= 35 and 500 <= variance <= 1500:
            is_biodegradable = True
            confidence = 0.80
            material_type = "Wood/Paper/Cardboard"
        else:
            is_biodegradable = False
            confidence = 0.60
            material_type = "Unknown Material"
        
        return {
            'is_biodegradable': is_biodegradable,
            'confidence': confidence,
            'material_type': material_type,
            'visual_features': {
                'hue': float(avg_hue),
                'saturation': float(avg_sat),
                'brightness': float(brightness),
                'texture_variance': float(variance),
                'edge_density': float(edge_density),
                'contour_count': num_contours
            }
        }
        
    except Exception as e:
        logger.error(f"ERROR: Visual analysis failed: {e}")
        return {'is_biodegradable': False, 'confidence': 0.5, 'material_type': 'Unknown'}

# ============================================================================
# FALLBACK VISUAL CLASSIFICATION
# ============================================================================
def fallback_visual_classify(image_path):
    logger.info("FALLBACK: Using visual analysis...")
    
    try:
        visual_result = advanced_visual_analysis(image_path)
        result = {
            'success': True,
            'classification': 'biodegradable' if visual_result['is_biodegradable'] else 'non-biodegradable',
            'confidence': visual_result['confidence'],
            'material_type': visual_result['material_type'],
            'method': 'visual_analysis_fallback',
            'details': visual_result.get('visual_features', {}),
            'recommendations': get_recommendations(visual_result['is_biodegradable'])
        }
        
        logger.info(f"SUCCESS: Fallback classification: {result['classification']} ({result['confidence']*100:.1f}%)")
        return result
        
    except Exception as e:
        logger.error(f"ERROR: Fallback failed: {e}")
        return {
            'success': False,
            'error': str(e),
            'classification': 'unknown',
            'confidence': 0.0
        }

# MAIN CLASSIFICATION FUNCTION
# ============================================================================
def classify_waste(image_path):

    result['image_path'] = image_path
    return result

# OUTPUT IMAGE GENERATION
# ============================================================================
def generate_output_image(image_path, result):
    try:
        image = cv2.imread(image_path)
        height, width = image.shape[:2]
        
        if result['classification'] == 'biodegradable':
            color = (0, 255, 0)
            bg_color = (0, 200, 0)
        else:
            color = (0, 0, 255)
            bg_color = (0, 0, 200)
        
        cv2.rectangle(image, (0, 0), (width, 100), bg_color, -1)
        
        text = f"{result['classification'].upper()}"
        confidence_text = f"Confidence: {result['confidence']*100:.1f}%"
        method_text = f"Method: {result['method']}"
        
        cv2.putText(image, text, (20, 35), cv2.FONT_HERSHEY_SIMPLEX, 1.0, (255, 255, 255), 3)
        cv2.putText(image, confidence_text, (20, 65), cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255, 255, 255), 2)
        cv2.putText(image, method_text, (20, 90), cv2.FONT_HERSHEY_SIMPLEX, 0.5, (255, 255, 255), 2)
        
        output_filename = f"classified_{os.path.basename(image_path)}"
        output_path = os.path.join(app.config['OUTPUT_FOLDER'], output_filename)
        cv2.imwrite(output_path, image)
        
        return output_filename
        
    except Exception as e:
        logger.error(f"ERROR: Output generation failed: {e}")
        return None

# PDF REPORT GENERATION
# ============================================================================
def generate_pdf_report(data_type='single', single_result=None):
    """Generate comprehensive PDF report"""
    try:
        buffer = io.BytesIO()
        doc = SimpleDocTemplate(buffer, pagesize=A4, rightMargin=30, leftMargin=30, topMargin=30, bottomMargin=18)
        
        story = []
        styles = getSampleStyleSheet()
        
        # Custom styles
        title_style = ParagraphStyle(
            'CustomTitle',
            parent=styles['Heading1'],
            fontSize=24,
            textColor=colors.HexColor('#667eea'),
            spaceAfter=30,
            alignment=TA_CENTER
        )
        
        heading_style = ParagraphStyle(
            'CustomHeading',
            parent=styles['Heading2'],
            fontSize=16,
            textColor=colors.HexColor('#764ba2'),
            spaceAfter=12
        )
        
        # Title
        title = Paragraph("AI Biodegradable Waste Classification Report", title_style)
        story.append(title)
        story.append(Spacer(1, 12))
        
        # Generation info
        info_text = f"<b>Generated:</b> {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}<br/>"
        info_text += f"<b>AI Model:</b> MobileNetV2 (TensorFlow/Keras)<br/>"
        info_text += f"<b>Report Type:</b> {'Single Classification' if data_type == 'single' else 'Complete History'}"
        story.append(Paragraph(info_text, styles['Normal']))
        story.append(Spacer(1, 20))
        
        if data_type == 'single' and single_result:
            # Single classification report
            story.append(Paragraph("Classification Details", heading_style))
            
            classification_color = colors.green if single_result['classification'] == 'biodegradable' else colors.red
            
            data = [
                ['Parameter', 'Value'],
                ['Classification', single_result['classification'].upper()],
                ['Confidence', f"{single_result['confidence']*100:.1f}%"],
                ['Material Type', single_result.get('material_type', 'Unknown')],
                ['Detection Method', single_result.get('method', 'N/A')],
                ['Timestamp', single_result.get('timestamp', 'N/A')]
            ]
            
            table = Table(data, colWidths=[3*inch, 3*inch])
            table.setStyle(TableStyle([
                ('BACKGROUND', (0, 0), (-1, 0), colors.HexColor('#667eea')),
                ('TEXTCOLOR', (0, 0), (-1, 0), colors.whitesmoke),
                ('ALIGN', (0, 0), (-1, -1), 'LEFT'),
                ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'),
                ('FONTSIZE', (0, 0), (-1, 0), 12),
                ('BOTTOMPADDING', (0, 0), (-1, 0), 12),
                ('BACKGROUND', (0, 1), (0, -1), colors.HexColor('#f0f0f0')),
                ('GRID', (0, 0), (-1, -1), 1, colors.black)
            ]))
            
            story.append(table)
            story.append(Spacer(1, 20))
            
            # Recommendations
            if 'recommendations' in single_result and single_result['recommendations']:
                story.append(Paragraph("Disposal Recommendations", heading_style))
                recs = single_result['recommendations']
                
                rec_text = f"<b>Disposal Method:</b> {recs.get('disposal_method', 'N/A')}<br/>"
                rec_text += f"<b>Bin Color:</b> {recs.get('bin_color', 'N/A')}<br/>"
                rec_text += f"<b>Decomposition Time:</b> {recs.get('decomposition_time', 'N/A')}<br/>"
                rec_text += f"<b>Environmental Impact:</b> {recs.get('environmental_impact', 'N/A')}<br/><br/>"
                rec_text += "<b>Tips:</b><br/>"
                for tip in recs.get('tips', []):
                    rec_text += f"• {tip}<br/>"
                
                story.append(Paragraph(rec_text, styles['Normal']))
        
        else:
            # Complete history report
            story.append(Paragraph("Classification History", heading_style))
            
            if len(CLASSIFICATION_HISTORY) > 0:
                # Summary statistics
                bio_count = sum(1 for h in CLASSIFICATION_HISTORY if h['classification'] == 'biodegradable')
                non_bio_count = len(CLASSIFICATION_HISTORY) - bio_count
                
                stats_text = f"<b>Total Classifications:</b> {len(CLASSIFICATION_HISTORY)}<br/>"
                stats_text += f"<b>Biodegradable:</b> {bio_count} ({bio_count/len(CLASSIFICATION_HISTORY)*100:.1f}%)<br/>"
                stats_text += f"<b>Non-Biodegradable:</b> {non_bio_count} ({non_bio_count/len(CLASSIFICATION_HISTORY)*100:.1f}%)"
                story.append(Paragraph(stats_text, styles['Normal']))
                story.append(Spacer(1, 20))
                
                # History table
                table_data = [['#', 'Filename', 'Classification', 'Confidence', 'Timestamp']]
                
                for idx, item in enumerate(CLASSIFICATION_HISTORY[-20:], 1):  # Last 20 items
                    table_data.append([
                        str(idx),
                        item['filename'][:30],
                        item['classification'],
                        f"{item['confidence']*100:.1f}%",
                        item['timestamp'][:19]
                    ])
                
                table = Table(table_data, colWidths=[0.5*inch, 2*inch, 1.5*inch, 1*inch, 1.5*inch])
                table.setStyle(TableStyle([
                    ('BACKGROUND', (0, 0), (-1, 0), colors.HexColor('#667eea')),
                    ('TEXTCOLOR', (0, 0), (-1, 0), colors.whitesmoke),
                    ('ALIGN', (0, 0), (-1, -1), 'CENTER'),
                    ('FONTNAME', (0, 0), (-1, 0), 'Helvetica-Bold'),
                    ('FONTSIZE', (0, 0), (-1, 0), 10),

                    ('ROWBACKGROUNDS', (0, 1), (-1, -1), [colors.white, colors.HexColor('#f0f0f0')])
                ]))
                
                story.append(table)
            else:
                story.append(Paragraph("No classification history available.", styles['Normal']))
        
        # Build PDF
        doc.build(story)
        buffer.seek(0)
        return buffer
        
    except Exception as e:
        logger.error(f"ERROR: PDF generation failed: {e}")
        traceback.print_exc()
        return None

@app.route('/api/classify', methods=['POST'])
def classify_image():
    try:
        if 'file' not in request.files:
            return jsonify({'success': False, 'error': 'No file provided'}), 400
        
        file = request.files['file']
        
        if file.filename == '':
            return jsonify({'success': False, 'error': 'No file selected'}), 400
        
        if not allowed_file(file.filename):
            return jsonify({'success': False, 'error': 'Invalid file type'}), 400
        
        filename = secure_filename(file.filename)
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        filename = f"{timestamp}_{filename}"
        filepath = os.path.join(app.config['UPLOAD_FOLDER'], filename)
        file.save(filepath)
        
        logger.info(f"FILE: Uploaded {filename}")
        
        result = classify_waste(filepath)
        
        add_to_history({
            'filename': filename,
            'classification': result['classification'],
            'confidence': result['confidence'],
            'timestamp': result['timestamp'],
            'method': result['method']
        })
        
        output_filename = generate_output_image(filepath, result)
        result['output_image'] = output_filename
        result['input_image'] = filename
        
        return jsonify(result)
        
    except Exception as e:
        logger.error(f"ERROR: Classification failed: {e}")
        traceback.print_exc()
        return jsonify({'success': False, 'error': str(e)}), 500

@app.route('/api/export/pdf', methods=['POST'])
def export_pdf():
    """Export classification report as PDF"""
    try:
        data = request.get_json()
        report_type = data.get('type', 'history')  # 'single' or 'history'
        single_result = data.get('result', None)
        
        logger.info(f"Generating PDF report: {report_type}")
        
        pdf_buffer = generate_pdf_report(
            data_type=report_type,
            single_result=single_result
        )
        
        if pdf_buffer is None:
            return jsonify({'success': False, 'error': 'PDF generation failed'}), 500
        
        filename = f"waste_classification_report_{datetime.now().strftime('%Y%m%d_%H%M%S')}.pdf"
        
        response = make_response(pdf_buffer.getvalue())
        response.headers['Content-Type'] = 'application/pdf'
        response.headers['Content-Disposition'] = f'attachment; filename={filename}'
        response.set_cookie('download_complete', 'true', max_age=5)
        
        logger.info(f"PDF report generated: {filename}")
        return response
        
    except Exception as e:
        logger.error(f"ERROR: PDF export failed: {e}")
        traceback.print_exc()
        return jsonify({'success': False, 'error': str(e)}), 500

                'non_biodegradable': sum(1 for h in CLASSIFICATION_HISTORY if h['classification'] == 'non-biodegradable')
            }
        }
        
        filename = f"waste_classification_data_{datetime.now().strftime('%Y%m%d_%H%M%S')}.json"
        
        response = make_response(json.dumps(data, indent=2))
        response.headers['Content-Type'] = 'application/json'
        response.headers['Content-Disposition'] = f'attachment; filename={filename}'
        response.set_cookie('download_complete', 'true', max_age=5)
        
        logger.info(f"JSON export completed: {filename}")
        return response
        
    except Exception as e:
        logger.error(f"ERROR: JSON export failed: {e}")
        return jsonify({'success': False, 'error': str(e)}), 500

@app.route('/api/export/excel', methods=['GET'])
def export_excel():
    """Export classification history as Excel"""
    try:
        if len(CLASSIFICATION_HISTORY) == 0:
            return jsonify({'success': False, 'error': 'No data to export'}), 400
        
        df = pd.DataFrame(CLASSIFICATION_HISTORY)
        
        output = io.BytesIO()
        with pd.ExcelWriter(output, engine='xlsxwriter') as writer:
            df.to_excel(writer, sheet_name='Classification History', index=False)
            
            workbook = writer.book
            worksheet = writer.sheets['Classification History']
            
            # Format header
            header_format = workbook.add_format({
                'bold': True,
                'bg_color': '#667eea',
                'font_color': 'white',
                'border': 1
            })
            
            for col_num, value in enumerate(df.columns.values):
                worksheet.write(0, col_num, value, header_format)
                worksheet.set_column(col_num, col_num, 20)
        
        output.seek(0)
        
        filename = f"waste_classification_data_{datetime.now().strftime('%Y%m%d_%H%M%S')}.xlsx"
        
        response = make_response(output.getvalue())
        response.headers['Content-Type'] = 'application/vnd.openxmlformats-officedocument.spreadsheetml.sheet'
        response.headers['Content-Disposition'] = f'attachment; filename={filename}'
        response.set_cookie('download_complete', 'true', max_age=5)
        
        logger.info(f"Excel export completed: {filename}")
        return response
        
    except Exception as e:
        logger.error(f"ERROR: Excel export failed: {e}")
        return jsonify({'success': False, 'error': str(e)}), 500
