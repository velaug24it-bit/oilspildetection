    <script>
        let map, markers = [], oilCount = 0, shipCount = 0, debrisCount = 0, alertCount = 0, fineTotal = 0;
        let uploadImg = null, autoInterval = null, satelliteMode = false;
        
        console.log('🌊 Initializing Ultimate Marine System...');
        
        // Initialize map
        map = L.map('map', {
            center: [20, 60],
            zoom: 3,
            minZoom: 2,
            maxZoom: 18,
            zoomControl: false
        });
        
        L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
            maxZoom: 19,
            attribution: '© OpenStreetMap | Ultimate Marine Detection'
        }).addTo(map);
        
        setTimeout(() => {
            map.invalidateSize();
            drawOceanZones();
            console.log('✅ Map loaded successfully');
            showToast('System initialized successfully', 'success');
        }, 200);
        
        function drawOceanZones() {
            const zones = [
                {name: 'North Atlantic Ocean', bounds: [[0,-80],[66,0]], color: '#4A90E2', center: [35,-40]},
                {name: 'Indian Ocean', bounds: [[-60,20],[30,120]], color: '#FFA94D', center: [-15,70]},
                {name: 'North Pacific Ocean', bounds: [[0,130],[66,-100]], color: '#38D9A9', center: [30,-170]},
                {name: 'South Pacific Ocean', bounds: [[-60,140],[0,-70]], color: '#20C997', center: [-25,-110]},
                {name: 'South Atlantic Ocean', bounds: [[-60,-60],[0,20]], color: '#5C7CFA', center: [-25,-20]},
                {name: 'Arctic Ocean', bounds: [[66,-180],[90,180]], color: '#A5D8FF', center: [75,0]},
                {name: 'Southern Ocean', bounds: [[-90,-180],[-60,180]], color: '#91A7FF', center: [-70,0]}
            ];
            
            zones.forEach(zone => {
                L.rectangle(zone.bounds, {
                    color: zone.color,
                    weight: 3,
                    fillOpacity: 0.08,
                    dashArray: '10,10'
                }).addTo(map).bindPopup(`<strong style="color:${zone.color};">${zone.name}</strong>`);
                
                L.marker(zone.center, {
                    icon: L.divIcon({
                        html: `<div style="background:${zone.color};color:white;padding:8px 16px;border-radius:20px;font-weight:700;font-size:13px;box-shadow:0 4px 15px rgba(0,0,0,0.5);border:2px solid white;">${zone.name}</div>`,
                        iconSize: [200,35],
                        className: ''
                    })
                }).addTo(map);
            });
        }
        
        function addMarker() {
            showLoading(true);
            setTimeout(() => {
                const lat = parseFloat(document.getElementById('lat').value);
                const lng = parseFloat(document.getElementById('lng').value);
                const type = document.getElementById('type').value;
                const severity = document.getElementById('severity').value;
                const confidence = document.getElementById('confidence').value;
                
                if (isNaN(lat) || isNaN(lng)) {
                    showToast('Invalid GPS coordinates!', 'danger');
                    showLoading(false);
                    return;
                }
                
                const colors = {oil:'#dc3545',ship:'#28a745',debris:'#17a2b8',algae:'#ffa500'};
                const icons = {oil:'oil-can',ship:'ship',debris:'trash',algae:'leaf'};
                const fines = {critical:5000000,high:3000000,medium:1500000,low:500000};
                const radii = {critical:8000,high:5000,medium:2500,low:1000};
                
                const fine = fines[severity] + Math.floor(Math.random() * 1000000);
                const color = colors[type];
                const icon = icons[type];
                
                const markerIcon = L.divIcon({
                    html: `<div style="position:relative;width:70px;height:70px;">
                        <div style="position:absolute;top:50%;left:50%;transform:translate(-50%,-50%);background:${color};width:55px;height:55px;border-radius:50%;display:flex;align-items:center;justify-content:center;color:white;border:4px solid white;box-shadow:0 6px 30px rgba(0,0,0,0.6);animation:markerPulse 2.5s infinite;">
                            <i class="fas fa-${icon}" style="font-size:26px;"></i>
                        </div>
                        ${type==='oil'?`<div style="position:absolute;top:50%;left:50%;transform:translate(-50%,-50%);width:55px;height:55px;border-radius:50%;animation:ripple 2.5s infinite;"></div>`:''}
                    </div>`,
                    iconSize: [70,70],
                    iconAnchor: [35,35]
                });
                
                const marker = L.marker([lat,lng], {icon:markerIcon}).addTo(map);
                
                if (type==='oil' || type==='algae') {
                    L.circle([lat,lng], {
                        radius: radii[severity],
                        color: color,
                        fillColor: color,
                        fillOpacity: 0.15,
                        weight: 2,
                        dashArray: '8,12'
                    }).addTo(map);
                }
                
                marker.bindPopup(`
                    <div style="min-width:350px;font-family:Poppins;">
                        <div style="background:linear-gradient(135deg,${color},${color}dd);color:white;padding:18px;margin:-15px -20px 15px -20px;border-radius:12px 12px 0 0;">
                            <h3 style="margin:0;font-size:20px;"><i class="fas fa-${icon}"></i> ${type.toUpperCase()} DETECTED</h3>
                        </div>
                        ${uploadImg?`<img src="${uploadImg}" style="width:100%;border-radius:12px;margin:12px 0;box-shadow:0 4px 20px rgba(0,0,0,0.3);">`:''}
                        <div style="background:#f8f9fa;padding:16px;border-radius:12px;margin:12px 0;border-left:4px solid ${color};">
                            <p style="margin:6px 0;"><strong>📍 GPS:</strong> ${lat.toFixed(6)}°, ${lng.toFixed(6)}°</p>
                            <p style="margin:6px 0;"><strong>🎯 Precision:</strong> ±5 meters</p>
                            <p style="margin:6px 0;"><strong>⏰ Time:</strong> ${new Date().toLocaleString()}</p>
                            <p style="margin:6px 0;"><strong>📊 Confidence:</strong> ${confidence}%</p>
                            <p style="margin:6px 0;"><strong>⚠️ Severity:</strong> <span style="color:${color};font-weight:700;">${severity.toUpperCase()}</span></p>
                        </div>
                        ${type==='oil'?`
                            <div style="background:#fff3cd;padding:16px;border-radius:12px;margin:12px 0;border-left:4px solid #ffa500;">
                                <h4 style="color:#856404;margin:0 0 12px 0;"><i class="fas fa-flag"></i> Responsible Authority</h4>
                                <p style="margin:6px 0;"><strong>🏴 India Coast Guard</strong></p>
                                <p style="margin:6px 0;font-size:13px;">Mumbai Regional Command</p>
                            </div>
                            <div style="background:#f8d7da;padding:16px;border-radius:12px;margin:12px 0;border-left:4px solid #dc3545;">
                                <h4 style="color:#721c24;margin:0 0 12px 0;"><i class="fas fa-dollar-sign"></i> Environmental Fine</h4>
                                <p style="font-size:32px;font-weight:800;color:#dc3545;margin:12px 0;">$${fine.toLocaleString()}</p>
                            </div>
                            <div style="background:#f8d7da;padding:16px;border-radius:12px;border-left:4px solid #dc3545;">
                                <p style="color:#721c24;font-weight:bold;margin:0;font-size:16px;">⛔ SHIP PERMIT CANCELLED</p>
                                <p style="margin:8px 0 0 0;font-size:13px;color:#721c24;">License suspended immediately</p>
                            </div>
                        `:''}
                    </div>
                `).openPopup();
                
                markers.push({marker,type,lat,lng});
                map.setView([lat,lng], 8);
                
                if (type==='oil') { oilCount++; alertCount+=5; fineTotal+=fine; }
                else if (type==='ship') shipCount++;
                else if (type==='debris') debrisCount++;
                
                updateStats();
                showLoading(false);
                showToast(`${type.toUpperCase()} detected successfully!`, type==='oil'?'danger':'success');
                
                document.getElementById('lat').value = (lat+(Math.random()-0.5)*0.01).toFixed(6);
                document.getElementById('lng').value = (lng+(Math.random()-0.5)*0.01).toFixed(6);

            document.getElementById('lng').value = lng;
            addMarker();

        function randomCoords() {
            const locs = [[18.523456,71.467890],[35.234123,-39.876543],[30.456789,169.890123],[-23.456789,56.789012]];
            const loc = locs[Math.floor(Math.random()*locs.length)];
            document.getElementById('lat').value = loc[0];
            document.getElementById('lng').value = loc[1];
        }
        
        function handleImg(e) {
            const file = e.target.files[0];
            if (file) {
                const reader = new FileReader();
                reader.onload = event => {
                    uploadImg = event.target.result;
                    showToast('Image uploaded successfully', 'success');
                };
                reader.readAsDataURL(file);
            }
        }
        
        function centerMap() {
            map.setView([20,60], 3);
            showToast('Map centered', 'info');
        }
        
        function autoDetection() {
            if (!autoInterval) {
                autoInterval = setInterval(randomCoords, 6000);
                showToast('Auto-detection: ON (every 6s)', 'success');
            } else {
                clearInterval(autoInterval);
                autoInterval = null;
                showToast('Auto-detection: OFF', 'info');
            }
        }
        
        function exportMapPNG() {
            showToast('Exporting map to PNG...', 'info');
            setTimeout(() => {
                html2canvas(document.getElementById('map'), {
                    useCORS: true,
                    allowTaint: true,
                    logging: false
                }).then(canvas => {
                    const link = document.createElement('a');
                    link.download = `marine-detection-${Date.now()}.png`;
                    link.href = canvas.toDataURL('image/png');
                    link.click();
                    showToast('Map exported successfully!', 'success');
                });
            }, 500);
        }
        
        function exportData() {
            const data = markers.map(m => ({
                type: m.type,
                lat: m.lat,
                lng: m.lng,
                time: new Date().toISOString()
            }));
            const blob = new Blob([JSON.stringify(data, null, 2)], {type: 'application/json'});
            const url = URL.createObjectURL(blob);
            const link = document.createElement('a');
            link.href = url;
            link.download = `marine-data-${Date.now()}.json`;
            link.click();
            URL.revokeObjectURL(url);
            showToast(`Exported ${markers.length} detections`, 'success');
        }
        
        function clearAll() {
            if (confirm('⚠️ Clear all detection markers?')) {
                markers.forEach(m => map.removeLayer(m.marker));
                markers = [];

                updateStats();
                showToast('All detections cleared', 'success');
            }
        }
